% Define acronyms using:
%   \newacronym{label}{SHORT}{Long form}
%
% Example:
%   \newacronym{rna}{RNA}{Ribonucleic acid}
%
%  Use the following commands in the text:
%
% \gls{rna}       → First use: "Ribonucleic acid (RNA)"
%                    Later uses: "RNA"
%                    Recommended for most cases (auto-expands & hyperlinks)
%
% \Gls{rna}       → Same as \gls but with capitalised first letter.
%                    Use at sentence start.
%
% \glspl{rna}     → Plural form: "Ribonucleic acids (RNAs)" / "RNAs"
% \Glspl{rna}     → Plural + capitalised
%
% ----------------------------------------------------------
%  Fixed forms (do NOT change after first use)
%
% \acrshort{rna}  → Always "RNA"
% \acrlong{rna}   → Always "Ribonucleic acid"
% \acrfull{rna}   → Always "Ribonucleic acid (RNA)"
%
% \acrshortpl{rna} → Always "RNAs"
% \acrlongpl{rna}  → Always "Ribonucleic acids"
% \acrfullpl{rna}  → Always "Ribonucleic acids (RNAs)"
%
% ----------------------------------------------------------
%  Linking behaviour
%
% - \gls, \Gls, \glspl, \Glspl → automatically link to glossary
% - \acr... commands → static text (no hyperlink)
%
% ----------------------------------------------------------
%  Organisation tip:
% - Keep all acronym definitions in a separate file (e.g., acronyms.tex)
%   and include it with:
%       \input{acronyms}
%
% - To force all acronyms to appear in the list (even if unused):
%       \glsaddall
