% \Glspl{rna}     → Plural + capitalised
%
% ----------------------------------------------------------
%  Fixed forms (do NOT change after first use)
%
% \acrshort{rna}  → Always "RNA"
% \acrlong{rna}   → Always "Ribonucleic acid"
% \acrfull{rna}   → Always "Ribonucleic acid (RNA)"
%
% \acrshortpl{rna} → Always "RNAs"
% \acrlongpl{rna}  → Always "Ribonucleic acids"
% \acrfullpl{rna}  → Always "Ribonucleic acids (RNAs)"
%
% ----------------------------------------------------------
%  Linking behaviour
%

% - \acr... commands → static text (no hyperlink)
